saisir entre

	si entre = 100
		alors, afficher "Arrêt immédiat du programme, explosion de votre machine imminente !"
	sinon,
		max <- entre
		min <- entre
		posPremMax <- 1

		tant que entre =/= 100

			pos <- pos + 1

			si entre > max
				alors, max <- entre
				posPremMax <- pos
				posDernMax <- pos
				occMax <- 1
			sinon si entre = max
					alors, occMax <- occMax + 1
			  		posDernMax <- pos
			  		fin si
			fin si

			si entre < min
				alors, min <- entre
				occMin <- 1
			sinon si entre = min
				alors, occMin <- occMin + 1
				fin si
			fin si

		afficher "Veuillez saisir un nouveau chiffre."
		saisir entre

		fin tant que

		afficher "le plus grand chiffre entré est: ", max
		afficher "le plus petit chiffre entré est: ", min
		afficher "la position du premier max est: ", posPremMax
		afficher "la position du dernier max est: ", posDernMax
		afficher "le nombre d'occurences du plus grand chiffre saisi est: ", occMax
		afficher "le nombre d'occurences du plus petit chiffre saisi est: ", occMin
