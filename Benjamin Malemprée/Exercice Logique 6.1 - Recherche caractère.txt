début exercice 6.1

début variables locales

var texte: T <- ""
var caractere: C <- ""
var premPos: N <- 0
var dernPos: N <- 0
var scan: N <- 1
var rep: N <- 0

fin variables locales

afficher "Veuillez saisir un texte et un caractère."
saisir texte
saisir caractere

	tant que scan < longueur(texte) + 1

		si caract(texte, scan) = caractere

			alors, dernPos <- scan
			rep <- rep + 1

			si prePos = 0

				alors, premPos <- scan

			fin si

		fin si

		scan <- scan + 1

	fin tant que

si rep != 0

	alors, Afficher "La première position où le caractère que vous avez saisi se trouve dans votre texte est : ", premPos
	Afficher "La dernière position où le caractère que vous avez saisi se trouve dans votre texte est : ", dernPos
	Afficher "Le nombre de fois que le caractère que vous avez saisi se trouve dans votre texte est : ", rep

sinon

	Afficher "Le caractère que vous avez saisi n'apparaît jamais dans votre texte"

fin si

fin exercice 6.1
