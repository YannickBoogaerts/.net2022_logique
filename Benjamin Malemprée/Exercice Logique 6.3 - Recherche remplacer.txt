var baseText: T <- ""
var searchText: T <- ""
var replacement: T <- ""
var scan: N <- 1
var replaced: B <- FAUX

fin variables locales

afficher "Veuillez saisir un texte source"
saisir baseText
afficher "Veuillez saisir un texte à rechercher"
saisir searchText
afficher "Veuillez saisir un texte de remplacement"
saisir replacement

si longueur(searchText) > longueur(baseText)

	alors, 
	Afficher "Recherche impossible dans votre texte. Le texte que vous cherchez est trop long"

	sinon,
	Tant que longueur(searchText) + scan <= longueur(baseText) + 1

		si sousChaine(baseText, scan, longueur(searchText) + scan) = searchText

			alors, 
			baseText <- sousChaine(baseText, 1, scan) (+) replacement (+) sousChaine(baseText, longueur(replacement) + scan, longueur(baseText) + 1)
 			replaced <- VRAI

		fin si

		si replaced = FAUX

			alors, 
			scan <- scan + 1

			sinon, 
			scan <- scan + longueur(replacement)
			replaced <- FAUX

		fin si

	fin tant que

	afficher baseText
